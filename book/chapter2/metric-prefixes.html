namespace mitama {

Usage 

// begin example
#include <dimensional/quantity.hpp>
#include <dimensional/si/all.hpp>
#include <dimensional/prefix.hpp>

int main() {
    using mitama::quantity_t;
    namespace si = mitama::si;

    quantity_t a = (1 | mitama::giga * si::meters); // 1 [Gm]
}
// end example
